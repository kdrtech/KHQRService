<script>
    // Initialize the module object
    var apiKhqrModule = new ApiKhqrModule();

    // Function to generate a Deeplink using async logic
    async function generateDeeplink() {
        try {
            // Data object containing necessary and optional parameters
            const data = {
                mobileNumber: "85598791802", // Optional: Customer's mobile number
                storeLabel: "ShipGlow",     // Optional: Store or merchant label
                amount: 1000,               // Required: Amount to process
                terminalLabel: "ShipGlow",  // Optional: Terminal label for identification
                currency: "KHR"             // Required: Transaction currency
            };

            // Generate the Deeplink by calling an API method (pseudo-code)
            const deeplinkResponse = await apiKhqrModule.createDeeplink(data);

            // Handle the response or display the generated link
            console.log("Generated DeepLink:", deeplinkResponse);
            alert(`Deeplink created successfully: ${deeplinkResponse}`);
        } catch (error) {
            // Error handling in case of failure
            console.error("Error generating DeepLink:", error);
            alert(`Failed to create DeepLink: ${error.message}`);
        }
    }

    // Example of calling the function (you can trigger this via an event)
    generateDeeplink();
</script>
<button onclick="generateDeeplink()">Generate Deeplink</button>
